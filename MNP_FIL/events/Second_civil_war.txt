add_namespace = Second_civil_war


country_event = {
	id = Second_civil_war.1
	title = Second_civil_war.1.t
	desc = Second_civil_war.1.d
	
	is_triggered_only = yes
	immediate = {
		set_nationality = {
			target_country = NMN
			character = EQU_Fleetfoot
		}
		
	}

	
	option={
		name = Second_civil_war.1.tt
		EQU = {
			EQU_Fleetfoot = {
				set_nationality = NWN
			}
	}
		if = {
			limit = {
				EQU_Fleetfoot = {
					is_corps_commander = yes
				}
			}
			EQU_Fleetfoot = {
				promote_leader = yes
			}
		}
		NWN = {
			set_cosmetic_tag = NWN_changeling_civil_war
			declare_war_on = { target = CLP type = civil_war }
			declare_war_on = { target = MAP type = civil_war }
			declare_war_on = { target = BAP type = civil_war }
			add_country_leader_role = {
				character = EQU_Fleetfoot
				promote_leader = yes
				country_leader = {
					ideology = despotism_military_junta
					expire = "1145.1.1.1"
					traits = {  }
				}
			}
		}
		
    }
	
}
