		selectable_mission = yes

			add_political_power = -50

		}
	}

				}
			}
		}
		days_remove = 15
		days_re_enable = 150
		icon = mission_icon
		is_good = yes
		selectable_mission = yes
		
		complete_effect = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -1000
			}
			CLK = {
				CLK = { add_to_variable = { CLK_Thorax_rebellion = 2 } }
				CLK = { add_to_variable = { CLK_Thorax_rebellion_stability_factor = -0.006 } }

					has_completed_focus = EER_C2s_3_10
					OR = {
						has_equipment = {
							artillery_equipment_1 > 500
						}
						has_equipment = {
							artillery_equipment_2 > 500
						}
						has_equipment = {
							artillery_equipment_3 > 500
						}
					}
				}
			}
		}
		days_remove = 15
		days_re_enable = 180
		icon = mission_icon
		is_good = yes
		selectable_mission = yes
		
		complete_effect = {
			if = {
				limit = {
					has_equipment = {
						artillery_equipment_1 > 500
					}
				}
				add_equipment_to_stockpile = {
					type = artillery_equipment_1
					amount = -500
				}
			}
			else_if = {
				limit = {
					has_equipment = {
						artillery_equipment_2 > 500
					}
				}
				add_equipment_to_stockpile = {
					type = artillery_equipment_2
					amount = -500
				}
			}
			else = {
				limit = {
					has_equipment = {
						artillery_equipment_3 > 500
					}
				}
				add_equipment_to_stockpile = {
					type = artillery_equipment_3
					amount = -500
				}
			}
			CLK = { add_to_variable = { CLK_Thorax_rebellion = 3 } }
            CLK = { add_to_variable = { CLK_Thorax_rebellion_stability_factor = -0.009 } }
            CLK = { add_to_variable = { CLK_Thorax_rebellion_war_support_factor  = -0.009 } }
