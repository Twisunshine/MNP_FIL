	on_startup = {
		effect = {
			CLK = {
				set_variable = { CLK_Thorax_rebellion = 60 }
				calculate_thorax_rebellion = yes
				
				set_variable = { CLK_queenist_faction_power = 5 }
				set_variable = { CLK_militarist_faction_power = 25 }
				set_variable = { CLK_royalist_faction_power = 30 }
				set_variable = { CLK_conservativist_faction_power = 20 }
				set_variable = { CLK_officer_faction_power = 15 }
				set_variable = { CLK_practitioner_faction_power = 5 }
				
				set_variable = { CLK_queen_faction_loyalty = 100 }
				set_variable = { CLK_militarist_faction_loyalty = 80 }
				set_variable = { CLK_royalist_faction_loyalty = 75 }
				set_variable = { CLK_conservativist_faction_loyalty = 65 }
				set_variable = { CLK_officer_faction_loyalty = 75 }
				set_variable = { CLK_practitioner_faction_loyalty = 60 }
				
				set_variable = { CLK_queen_value = 0 }
				
				set_variable = { CLK_factions_selected = 1 }
				
				CLK_calculate_queen_value_debuff = yes
				CLK_calculate_factions_debuff = yes
				add_dynamic_modifier = { modifier = CLK_Factions_dynamic_modifier }
			}
		}
	}
	
	on_daily_CLK = {
		effect = { 
			CLK_calculate_sur_controll = yes
		}
	}
	
	on_weekly_CLK = { 
		effect = {
			CLK_calculate_queen_value_debuff = yes
			CLK_calculate_factions_debuff = yes
		}
	}
